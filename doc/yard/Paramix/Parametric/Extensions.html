def append_features(base)
  return super(base) if Mixin === base || Mixin === self

  base.extend(Extensions)

  anc = ancestors.find{ |a| a.respond_to?(:parameterized) }
  base.parameterized.concat(anc.parameterized)

  super(base)
end
